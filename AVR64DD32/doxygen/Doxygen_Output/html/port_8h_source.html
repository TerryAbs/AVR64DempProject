121static inline void PORTA_set_port_dir(const uint8_t mask, const enum port_dir dir)
122{
123 switch (dir) {
124 case PORT_DIR_IN:
125 VPORTA.DIR &= ~mask;
126 break;
127 case PORT_DIR_OUT:
128 VPORTA.DIR |= mask;
129 break;
130 case PORT_DIR_OFF:
131 /*/ should activate the pullup for power saving
132 but a bit costly to do it here */
133 {
134 for (uint8_t i = 0; i < 8; i++) {
135 if (mask & 1 << i) {
136 *((uint8_t *)&PORTA + 0x10 + i) |= 1 << PORT_PULLUPEN_bp;
137 }
138 }
139 }
140 break;
141 default:
142 break;
143 }
144}
